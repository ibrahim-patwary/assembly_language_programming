;ADDITION OF TWO NUMBERS
;SIMPLE INPUT-OUTPUT PROGRAM
.MODEL SMALL
.STACK 100H  

.DATA

NUM1 DB 4
NUM2 DB ? 
  
.CODE

MAIN PROC
    
    MOV AX,@DATA
    MOV DS, AX
              
    MOV AH,1
    INT 21H
    MOV NUM2,AL
    
    MOV AH,2
    ADD NUM1,48
    MOV DL, NUM1
    INT 21H
    
    MOV DL, NUM2
    INT 21H             

    MOV AH, 4CH
    INT 21H
    MAIN ENDP

END MAIN
